// Wait for SDK to load
async function waitForSDK() {
  return new Promise(resolve => {
    if (typeof window.mcj !== 'undefined') {
      resolve();
    } else {
      setTimeout(() => waitForSDK().then(resolve), 100);
    }
  });
}

// Get user's wallet address
await waitForSDK();
